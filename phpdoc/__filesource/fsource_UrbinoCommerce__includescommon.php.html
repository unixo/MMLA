  83. /**
  84.  * Wrapper per interfacciarsi al database tramite istruzioni SQL
  85.  * 
  86.  * E' possibile effettuare una SELECT o una DML (UPDATE, INSERT). All'interno
  87.  * della funzione viene fatto uso delle classi DB2 (PEAR) per creare lo
  88.  * statement e, eventualmente, assegnare i valori alle rispettive
  89.  * 'bind variables'.
  90.  * L'uso di questi accorgimenti previene possibili attacchi di tipo SQL
  91.  * injection o XSS stored.
  92.  *
  93.  * Esempio d'utilizzo:
  94.  * <code>
  95.  * $results = executeQuery('SELECT * FROM tabella WHERE campo=?', $valore);
  96.  * echo 'Numero di record trovati: '.count($results);
  97.  * </code>
  98.  *
  99.  * @param string $sql    Comando SQL da eseguire
  100.  * @param array  $values Valore delle bind variables
  101.  *
  102.  * @return array Result set ottenuto dalla query (se fosse una SELECT)
  103.  */
  104. function executeQuery($sql, $values = array())
  105. {
  106.     global $db_conn;
  107.  
  108.     $results = array();
  109.     if (sizeof($values) > 0) {
  110.         $statement = $db_conn->prepare($sql, null, MDB2_PREPARE_RESULT);
  111.         $resultset = $statement->execute($values);
  112.         $statement->free();
  113.     } else {
  114.         $resultset = $db_conn->query($sql);
  115.     }
  116.     if (PEAR::isError($resultset) === true) {
  117.         throw new PEAR_Exception($resultset->getMessage(), ERR_DATABASE);
  118.     }
  119.  
  120.     if (substr(strtolower($sql), 0, 6) === 'select') {
  121.         while (($row = $resultset->fetchRow(MDB2_FETCHMODE_ASSOC)) !== null) {
  122.             $results[] = $row;
  123.         }
  124.     }
  125.     $resultset->free();
  126.  
  127.     return $results;
  128. }

  247. function getConfigValue($attrib)
  248. {
  249.     if (strstr($attrib, '%')) {
  250.         $sql     = 'SELECT * FROM config WHERE attrib LIKE ?';
  251.         $records = executeQuery($sql, $attrib);
  252.         foreach ($records as $row) {
  253.             $results[$row['attrib']] = $row['value'];
  254.         }
  255.         return $results;
  256.     } else {
  257.         $sql     = 'SELECT * FROM config WHERE attrib = ?';
  258.         $records = executeQuery($sql, $attrib);
  259.         return $records[0]['value'];
  260.     }
  261. }
