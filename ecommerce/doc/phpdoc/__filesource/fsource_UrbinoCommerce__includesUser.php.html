  349.     public static function checkEmail($email)
  350.     {
  351.         /**
  352.          * Prima di tutto controllo che sia presente il carattere '@' e che
  353.          * la lunghezza dell'indirizzo sia valida: sara' sufficiente effettuare
  354.          * un match con la regexp (estesa) che descrive un email

  356.         if (!ereg("^[^@]{1,64}@[^@]{1,255}$", $email)) {
  357.             return false;
  358.         }
  359.         /**
  360.          * Divido in due parti l'indirizzo: utente e dominio
  361.          */
  362.         $email_array = explode("@", $email);
  363.         $local_array = explode(".", $email_array[0]);
  364.         for ($i = 0; $i < sizeof($local_array); $i++) {
  365.             if (!ereg("^(([A-Za-z0-9!#$%&'*+/=?^_`{|}~-][A-Za-z0-9!#$%&'*+/=".
  366.                       "?^_`{|}~\.-]{0,63})|(\"[^(\\|\")]{0,62}\"))$", 
  367.                       $local_array[$i])) {
  368.                 return false;
  369.             }
  370.         }
  371.         /**
  372.          * Controllo se il dominio e' rappresentato da un indirizzo IP: se non
  373.          * lo e', controllo la correttezza della sintassi.
  374.          */
  375.         if (!ereg("^\[?[0-9\.]+\]?$", $email_array[1])) {
  376.             $domain_array = explode(".", $email_array[1]);
  377.             if (sizeof($domain_array) < 2) {
  378.                 /**
  379.                  * C'e' solo un punto: non puo' essere ne' un dominio corretto
  380.                  * né un indirizzo IP.
  381.                  */
  382.                 return false;
  383.             }
  384.             for ($i = 0; $i < sizeof($domain_array); $i++) {
  385.                 if (!ereg("^(([A-Za-z0-9][A-Za-z0-9-]{0,61}[A-Za-z0-9])|".
  386.                           "([A-Za-z0-9]+))$", $domain_array[$i])) {
  387.                     return false;
  388.                 }
  389.             }
  390.         }
  391.         return true;
  392.     }
